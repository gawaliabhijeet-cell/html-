    <div id="result" class="hidden">
        <h2 id="resultMessage"></h2>
    </div>

    <script>
        const yesButton = document.getElementById('yesButton');
        const noButton = document.getElementById('noButton');
        const resultDiv = document.getElementById('result');
        const resultMessage = document.getElementById('resultMessage');

        yesButton.addEventListener('click', function() {
            yesButton.textContent = "You said Yes!";
            noButton.textContent = "No, I changed my mind!";
            noButton.style.display = "inline-block"; // Show the no button
        });

        noButton.addEventListener('click', function() {
            resultDiv.classList.remove('hidden');
            resultMessage.textContent = "I love you too!";
            yesButton.style.display = "none"; // Hide the yes button
            noButton.style.display = "none"; // Hide the no button
        });
    </script>
